--  DDL for Function SFN_IS_AUTH_MAP
--------------------------------------------------------

  CREATE OR REPLACE FUNCTION "SFN_IS_AUTH_MAP" ( I_AUTH_CD     IN VARCHAR2
                                             , I_AUTH_DVCD   IN VARCHAR2
                                             , I_USER_DVCD   IN VARCHAR2
                                             , I_WORK_DVCD   IN VARCHAR2
                                             , I_STAT_DVCD   IN VARCHAR2
                                             , I_DEPT_CD     IN VARCHAR2)
RETURN VARCHAR2 DETERMINISTIC
/*
=====================================================================
  프로그램ID : SFN_IS_AUTH_MAP
  입  력  값 : 
              I_AUTH_CD     IN VARCHAR2 -- 입력코드
              I_AUTH_DVCD   IN VARCHAR2 -- 권한분류
              I_USER_DVCD   IN VARCHAR2 -- 사용자분류
              I_WORK_DVCD   IN VARCHAR2 -- 직업분류
              I_STAT_DVCD   IN VARCHAR2 -- 상태분류
  출  력  값 : Y / N
*********************************************************************
  수정일자     수정자    수정내용
=====================================================================
*/
IS
  V_RETURN_VALUE VARCHAR2(1) DEFAULT 'N';
  V_COUNT        NUMBER  DEFAULT 0;

BEGIN

    IF I_AUTH_CD = 'ALL' THEN
        V_RETURN_VALUE := 'Y';
    ELSIF I_AUTH_CD = 'NO_LOGIN' THEN
        V_RETURN_VALUE := 'Y';
    ELSE
        SELECT      COUNT(*)
        INTO        V_COUNT
        FROM        
        (
                    -- 개인 권한
                    SELECT  	CA.AUTH_CD
                    FROM		COM_AUTH CA, COM_AUTH_MAP CAM
                    WHERE		CA.AUTH_CD = CAM.AUTH_CD(+)
                    AND			CAM.AUTH_DVCD = I_AUTH_DVCD
                    AND			(
                                    CAM.USER_DVCD = I_USER_DVCD AND CAM.WORK_DVCD = I_WORK_DVCD AND CAM.STAT_DVCD = I_STAT_DVCD
                                    OR
                                    CAM.USER_DVCD = I_USER_DVCD AND CAM.WORK_DVCD = I_WORK_DVCD AND CAM.STAT_DVCD = 'ALL'
                                    OR
                                    CAM.USER_DVCD = I_USER_DVCD AND CAM.WORK_DVCD = 'ALL' AND CAM.STAT_DVCD = 'ALL'
                                    OR
                                    CAM.USER_DVCD = 'ALL' AND CAM.WORK_DVCD = 'ALL' AND CAM.STAT_DVCD = 'ALL'
                    )
                    GROUP BY 	CA.AUTH_CD
                    
                    UNION
                    
                    -- 부서 권한                    
                    SELECT  	CA.AUTH_CD
                    FROM		COM_AUTH CA, COM_AUTH_DEPT CAD
                    WHERE		CA.AUTH_CD = CAD.AUTH_CD(+)
                    AND			CAD.AUTH_DVCD = I_AUTH_DVCD
                    AND			CAD.DEPT_CD = I_DEPT_CD
                    GROUP BY 	CA.AUTH_CD
                    
        )
        WHERE       AUTH_CD = I_AUTH_CD;

        IF V_COUNT > 0 THEN
            V_RETURN_VALUE := 'Y';
        ELSE V_RETURN_VALUE := 'N';
        END IF;
    END IF;    
    RETURN V_RETURN_VALUE;
END SFN_IS_AUTH_MAP;
